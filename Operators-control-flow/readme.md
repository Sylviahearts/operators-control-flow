Arithmetic operators
+, -, *, /
Eg- let a = 3;
    let b = 4;

    console.log(a + b)
    console.log(a * b)


Assignment Operators
=, +=.-=, /=, *=
Eg- let a = 5
    let b = a + 2

Comparison Operators
<, >, <=, >=, !+, ==, ===
Eg- let x = 5;
console.log(x > 6)
console.log(x <= 5)

et EligibleVoter = Age || RegisteredVoter
